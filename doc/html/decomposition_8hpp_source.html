393  template<typename VariableCombinationFunction>
394  auto combine(
395  const decomposition& other,
396  VariableCombinationFunction combination) const
397  {
398  decomposition result;
399 
400  // compute union of the deps_ sets
401  result.deps_ = sorted_union(deps_, other.deps_);
402 
403  // compute number of values for the deps_ vector
404  std::size_t num_values = 1;
405  for (auto&& var : result.deps_)
406  {
407  num_values *= var->size();
408  }
409 
410  // determine which dependencies are in A and which are in B
411  const std::uint8_t A = 1;
412  const std::uint8_t B = 2;
413  std::vector<std::uint8_t> is_in;
414  auto left = deps_.begin();
415  auto right = other.deps_.begin();
416  for (;;)
417  {
418  if (left == deps_.end() && right == other.deps_.end())
419  {
420  break;
421  }
422  else if (left == deps_.end())
423  {
424  is_in.push_back(B);
425  ++right;
426  }
427  else if (right == other.deps_.end())
428  {
429  is_in.push_back(A);
430  ++left;
431  }
432  else if (*left < *right)
433  {
434  is_in.push_back(A);
435  ++left;
436  }
437  else if (*right < *left)
438  {
439  is_in.push_back(B);
440  ++right;
441  }
442  else // *right == *left
443  {
444  is_in.push_back(A | B);
445  ++left;
446  ++right;
447  }
448  }
449 
450  assert(is_in.size() == result.deps_.size());
451 
452  // compute the conditional random variables
453  for (std::size_t i = 0; i < num_values; ++i)
454  {
455  std::size_t index_a = 0;
456  std::size_t index_b = 0;
457 
458  // find corresponding variables in both trees
459  std::size_t index_result = i;
460  std::size_t size_a = 1;
461  std::size_t size_b = 1;
462  for (std::size_t j = 0; j < result.deps_.size(); ++j)
463  {
464  auto&& var = result.deps_[j];
465  auto var_values_count = var->size();
466  if ((is_in[j] & A) != 0)
467  {
468  index_a += (index_result % var_values_count) * size_a;
469  size_a *= var_values_count;
470  }
471  if ((is_in[j] & B) != 0)
472  {
473  index_b += (index_result % var_values_count) * size_b;
474  size_b *= var_values_count;
475  }
476  index_result /= var_values_count;
477  }
478 
479  // combine them
480  result.vars_.push_back(
481  combination(vars_[index_a], other.vars_[index_b]));
482  }
483 
484  return result;
485  }
