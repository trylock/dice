515  friend auto roll(
516  const random_variable& num_dice,
517  const random_variable& num_faces)
518  {
519  // If there are no dice or dice sizes,
520  // then this is an impossible event
521  if (num_dice.probability_.empty() ||
522  num_faces.probability_.empty())
523  {
524  return random_variable{};
525  }
526 
527  // Throw an exception if the number of dice or
528  // number of dice faces is not positive.
529  // note: Use the restrict method to restrict variable's
530  // range to positive integers.
531  for (auto&& pair : num_dice.probability_)
532  {
533  if (pair.first <= 0)
534  {
535  throw std::invalid_argument(
536  "Number of dice has to be a positive integer.");
537  }
538  }
539 
540  for (auto&& pair : num_faces.probability_)
541  {
542  if (pair.first <= 0)
543  {
544  throw std::invalid_argument(
545  "Number of dice faces has to be a positive integer.");
546  }
547  }
548 
549  // find maximal number of dice and faces
550  auto max_dice = num_dice.max_value();
551 
552  // compute distribution for each possible number of faces
553  random_variable dist;
554  for (auto&& pair : num_faces.probability_)
555  {
556  auto faces_count = pair.first;
557  auto faces_prob = pair.second;
558  auto base_prob = 1 / static_cast<probability_type>(faces_count);
559 
560  // save the probability of 1 roll
561  auto one_roll = num_dice.probability_.find(1);
562  if (one_roll != num_dice.probability_.end())
563  {
564  auto rolls_prob = one_roll->second;
565  auto prob = base_prob * faces_prob * rolls_prob;
566  for (value_type i = 1; i <= faces_count; ++i)
567  {
568  dist.add_probability(i, prob);
569  }
570  }
571 
572  // Prefix sum of probability:
573  // P(XdY = k | X = dice_count, Y = faces_count)
574  std::vector<probability_type> probability(
575  faces_count * max_dice + 1, 0);
576 
577  // base case: roll only 1 die
578  for (value_type i = 1; i <= faces_count; ++i)
579  {
580  probability[i] = base_prob;
581  }
582 
583  // Roll `dice_count` dice given the result of
584  // `dice_count - 1` dice
585  for (value_type dice_count = 2;
586  dice_count <= max_dice;
587  ++dice_count)
588  {
589  // compute the prefix sum of the probability array
590  for (value_type i = 2; i <= faces_count * dice_count; ++i)
591  {
592  probability[i] = probability[i - 1] + probability[i];
593  }
594 
595  // For computation of the probability of the sum of i
596  // we only need values j < i. By iterating backwards we
597  // don't overwrite those values.
598  for (auto i = faces_count * dice_count; i >= dice_count; --i)
599  {
600  // compute the probability of the sum of i
601  value_type j = std::max(i - faces_count, 1);
602  auto prob_i = probability[i - 1] - probability[j - 1];
603  prob_i *= base_prob;
604 
605  // We will break the invariant that the probability
606  // array is a prefix sum of the probabilities but it
607  // will be restored in the next iteration.
608  probability[i] = prob_i;
609  }
610 
611  // zero out probabilities of lower values
612  for (value_type i = 1; i < dice_count; ++i)
613  {
614  probability[i] = 0;
615  }
616 
617  // check whether dice_count is a valid number of dice
618  auto rolls_it = num_dice.probability_.find(dice_count);
619  if (rolls_it == num_dice.probability_.end())
620  {
621  continue;
622  }
623 
624  // save the probability
625  auto rolls_prob = rolls_it->second;
626  for (auto i = dice_count; i <= faces_count * dice_count; ++i)
627  {
628  dist.add_probability(
629  i,
630  probability[i] * rolls_prob * faces_prob
631  );
632  }
633  }
634  }
635  return dist;
636  }
